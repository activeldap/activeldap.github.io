def replace(entry)
  if entry.nil?
    @target = @owner[@options[:foreign_key_name]] = nil
  else
    @target = (Proxy === entry ? entry.target : entry)
    infect_connection(@target)
    unless entry.new_entry?
      @owner[@options[:foreign_key_name]] = entry[primary_key]
    end
    @updated = true
  end

  loaded
  entry
end
