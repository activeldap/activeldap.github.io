def initialize(name, schema, group)
  @schema = schema
  @name, *@aliases = attribute("NAME", name)
  @name ||= name
  @id = @schema.resolve_name(group, @name)
  collect_info
  @schema = nil
end

def eql?(other)
  self.class == other.class and
    (id == other.id or
     (id.nil? and other.nil? and name == other.name))
